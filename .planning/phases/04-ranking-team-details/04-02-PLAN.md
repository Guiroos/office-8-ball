**`src/components/ranking/type-tabs.tsx`**

`"use client"` directive required (uses useRouter/usePathname from next/navigation to update URL search params).

Named export `TypeTabs`. Props: `activeType: "solo" | "duo" | "all"`.

Two tab buttons: "Solo" and "Duplas". Plus an implicit "Todos" if needed — but per D-02, only "Solo" | "Duplas" tabs. Show both tabs; when neither matches, no tab is active.

```tsx
"use client";
import { useRouter, usePathname, useSearchParams } from "next/navigation";

export function TypeTabs({ activeType }: { activeType: "solo" | "duo" | "all" }) {
  const router = useRouter();
  const pathname = usePathname();
  const searchParams = useSearchParams();

  function setType(type: "solo" | "duo" | "all") {
    const params = new URLSearchParams(searchParams.toString());
    if (type === "all") {
      params.delete("type");
    } else {
      params.set("type", type);
    }
    router.push(`${pathname}?${params.toString()}`);
  }

  // Tab buttons: "Todos" | "Solo" | "Duplas"
  // Active tab: bg-primary text-primary-foreground; inactive: bg-surface-muted text-muted-foreground
  return (
    <div className="flex gap-2">
      {(["all", "solo", "duo"] as const).map((type) => (
        <button
          key={type}
          onClick={() => setType(type)}
          className={cn(
            "rounded-md px-4 py-2 text-sm font-medium transition-colors",
            activeType === type
              ? "bg-primary text-primary-foreground"
              : "bg-surface-muted text-muted-foreground hover:bg-surface-emphasis"
          )}
        >
          {type === "all" ? "Todos" : type === "solo" ? "Solo" : "Duplas"}
        </button>
      ))}
    </div>
  );
}
```

```tsx
import { PodiumCard } from "@/components/ranking/podium-card";
```

```tsx
import type { Metadata } from "next";
import { listAllTeamsWithStats } from "@/lib/ranking";
import { RankingView } from "@/components/ranking/ranking-view";

export const metadata: Metadata = {
  title: "Ranking | Office 8 Ball",
  description: "Classificação geral dos times do Office 8 Ball.",
};

type SearchParams = { type?: string };

export default async function RankingPage({
  searchParams,
}: {
  searchParams: Promise<SearchParams>;
}) {
  const { type } = await searchParams;
  const activeType =
    type === "solo" ? "solo" : type === "duo" ? "duo" : "all";

  const teams = await listAllTeamsWithStats(
    activeType === "all" ? undefined : activeType
  );

  return (
    <main className="container mx-auto px-4 py-8">
      <RankingView teams={teams} activeType={activeType} />
    </main>
  );
}
```

Note: `searchParams` is a Promise in Next.js 15+ App Router — must await it. This matches the `params` async pattern documented in STATE.md decision #8.
