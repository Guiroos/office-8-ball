```tsx
function teamInitials(name: string): string {
  return name
    .split(" ")
    .slice(0, 2)
    .map((w) => w[0]?.toUpperCase() ?? "")
    .join("");
}

// Deterministic color from team name hash (so same team always gets same color)
function avatarColor(name: string): string {
  const colors = [
    "bg-team-alpha-soft text-team-alpha",
    "bg-team-beta-soft text-team-beta",
    "bg-surface-success text-primary",
    "bg-gold-soft text-gold",
  ];
  let hash = 0;
  for (const char of name) hash = (hash * 31 + char.charCodeAt(0)) >>> 0;
  return colors[hash % colors.length];
}

export function TeamCard({ team }: { team: TeamRecord }) {
  return (
    <Link href={`/times/${team.id}`}>
      <div className="flex items-center gap-4 rounded-xl border bg-surface-emphasis p-4 shadow-sm shadow-foreground/8 hover:shadow-md transition-shadow cursor-pointer">
        {/* Avatar initials */}
        <div className={cn("flex h-12 w-12 shrink-0 items-center justify-center rounded-full text-lg font-semibold", avatarColor(team.name))}>
          {teamInitials(team.name)}
        </div>
        {/* Name + type */}
        <div className="flex flex-col gap-1 flex-1 min-w-0">
          <span className="body font-semibold truncate capitalize">{team.name}</span>
          <Badge variant="outline" className="caption w-fit">
            {team.type === "solo" ? "Solo" : "Duplas"}
          </Badge>
        </div>
      </div>
    </Link>

}
```

**`src/components/teams/recent-matches-list.tsx`**

Named export `RecentMatchesList`. Props: `matches: MatchRecord[]`, `teamId: string`, `allTeams: TeamRecord[]`.

Shows last 3 matches. `allTeams` is used to look up opponent name.

```tsx
import { SectionHeader } from "@/components/primitives/section-header";
import { Badge } from "@/components/ui/badge";
import type { MatchRecord, TeamRecord } from "@/lib/types";

export function RecentMatchesList({
  matches,
  teamId,
  allTeams,
}: {
  matches: MatchRecord[];
  teamId: string;
  allTeams: TeamRecord[];
}) {
  const recent = matches
    .filter((m) => m.teamAId === teamId || m.teamBId === teamId)
    .slice(0, 3);

  function opponentName(match: MatchRecord): string {
    const opponentId = match.teamAId === teamId ? match.teamBId : match.teamAId;
    return allTeams.find((t) => t.id === opponentId)?.name ?? opponentId.slice(0, 8);
  }

  function formatDate(iso: string): string {
    return new Intl.DateTimeFormat("pt-BR", { day: "2-digit", month: "short" }).format(new Date(iso));
  }

  return (
    <div className="flex flex-col gap-4">
      <SectionHeader eyebrow="Histórico" title="Últimas Partidas" />
      {recent.length === 0 ? (
        <p className="body text-muted-foreground">Nenhuma partida registrada</p>
      ) : (
        <ul className="flex flex-col gap-2">
          {recent.map((match) => {
            const isWin = match.winnerTeamId === teamId;
            return (
              <li key={match.id} className="flex items-center justify-between rounded-lg border bg-surface p-3">
                <span className="caption text-muted-foreground">{formatDate(match.playedAt)}</span>
                <span className="body">vs. <span className="capitalize">{opponentName(match)}</span></span>
                <Badge variant={isWin ? "default" : "outline"} className={isWin ? "bg-surface-success text-primary" : undefined}>
                  {isWin ? "V" : "D"}
                </Badge>
              </li>
            );
          })}
        </ul>
      )}
      <p className="caption text-muted-foreground">Ver histórico completo</p>
    </div>
  );
}
```

**`src/components/teams/h2h-section.tsx`**

`"use client"` directive required (uses useState for selected opponent).

Named export `H2HSection`. Props: `currentTeamId: string`, `rivals: TeamRecord[]`, `defaultRivalId: string | null`, `allMatches: MatchRecord[]`.

The parent RSC passes pre-computed `allMatches` down; H2HSection calls `computeHeadToHead` client-side when opponent changes (acceptable since allMatches is already loaded).

```tsx
"use client";
import { useState } from "react";
import { SectionHeader } from "@/components/primitives/section-header";
import { StatTile } from "@/components/primitives/stat-tile";
import { computeHeadToHead } from "@/lib/stats";
import type { MatchRecord, TeamRecord } from "@/lib/types";

export function H2HSection({
  currentTeamId,
  rivals,
  defaultRivalId,
  allMatches,
}: {
  currentTeamId: string;
  rivals: TeamRecord[];
  defaultRivalId: string | null;
  allMatches: MatchRecord[];
}) {
  const [selectedRivalId, setSelectedRivalId] = useState<string | null>(defaultRivalId);

  const selectedRival = rivals.find((t) => t.id === selectedRivalId) ?? null;
  const h2h = selectedRival
    ? computeHeadToHead(currentTeamId, selectedRival.id, allMatches)
    : null;

  // Last match date: find the most recent match in recentMatchIds
  const lastMatchDate = h2h?.recentMatchIds[0]
    ? allMatches.find((m) => m.id === h2h.recentMatchIds[0])?.playedAt
    : null;

  function formatDate(iso: string): string {
    return new Intl.DateTimeFormat("pt-BR", { day: "2-digit", month: "short", year: "numeric" }).format(new Date(iso));
  }

  return (
    <div className="flex flex-col gap-4">
      <SectionHeader eyebrow="Rivalidade" title="Confrontos Diretos" />

      {/* Opponent selector */}
      <div className="flex flex-col gap-2">
        <label className="caption text-muted-foreground">Selecione um adversário</label>
        <select
          value={selectedRivalId ?? ""}
          onChange={(e) => setSelectedRivalId(e.target.value || null)}
          className="rounded-md border bg-surface px-3 py-2 body text-foreground focus:outline-none focus:ring-2 focus:ring-primary"
        >
          <option value="">Selecione um adversário</option>
          {rivals.map((rival) => (
            <option key={rival.id} value={rival.id} className="capitalize">
              {rival.name}
            </option>
          ))}
        </select>
      </div>

      {/* H2H stats */}
      {selectedRival && h2h ? (
        h2h.totalMatches === 0 ? (
          <p className="body text-muted-foreground">Nenhuma partida entre estes times</p>
        ) : (
          <div className="grid grid-cols-2 gap-4 sm:grid-cols-4">
            <StatTile label="Vitórias Contra" value={h2h.teamAWins} />
            <StatTile label="Derrotas Para" value={h2h.teamBWins} />
            <StatTile
              label="Taxa vs Adversário"
              value={
                h2h.totalMatches === 0
                  ? "—"
                  : `${((h2h.teamAWins / h2h.totalMatches) * 100).toFixed(1)}%`
              }
            />
            <StatTile
              label="Última Partida"
              value={lastMatchDate ? formatDate(lastMatchDate) : "—"}
            />
          </div>
        )
      ) : null}
    </div>
  );
}
```

```tsx
  const { id } = await params;
  if (!hasDatabaseUrl()) return { title: "Times | Office 8 Ball" };
  const team = await getTeamById(id);
  return {
    title: team ? `${team.name} | Office 8 Ball` : "Time não encontrado | Office 8 Ball",
  };
}

export default async function TeamDetailPage({ params }: Props) {
  const { id } = await params;

  if (!hasDatabaseUrl()) notFound();

  const team = await getTeamById(id);
  if (!team) notFound();

  const session = await getServerSession(getAuthOptions());
  const userId = session?.user?.id ?? "";

  // Fetch all matches for the user's teams (scoped to user's visibility)
  const allMatches = await listMatches(userId);

  // Compute stats for this team
  const stats = computeTeamStats(team.id, allMatches);

  // Fetch rivals = all user teams excluding this team
  const userTeams = userId ? await listUserTeams(userId) : [];
  const rivals: TeamRecord[] = userTeams.filter((t) => t.id !== team.id);

  // Pre-load primary rival: team with most H2H matchups against current team
  let defaultRivalId: string | null = null;
  if (rivals.length > 0) {
    let maxMatches = 0;
    for (const rival of rivals) {
      const h2h = computeHeadToHead(team.id, rival.id, allMatches);
      if (h2h.totalMatches > maxMatches) {
        maxMatches = h2h.totalMatches;
        defaultRivalId = rival.id;
      }
    }
    // Fallback: if no H2H matches exist, pre-load first rival
    if (!defaultRivalId) defaultRivalId = rivals[0]?.id ?? null;
  }

  return (
    <main className="container mx-auto px-4 py-8">
      <TeamDetailView
        team={team}
        stats={stats}
        recentMatches={allMatches}
        allMatches={allMatches}
        rivals={rivals}
        defaultRivalId={defaultRivalId}
      />
    </main>
  );
}
```
