```typescript
import { prisma } from "@/lib/prisma";
import { computeTeamStats } from "@/lib/stats";
import type { MatchRecord, TeamRecord } from "@/lib/types";
import type { TeamStats } from "@/lib/stats";

function hasDatabaseUrl() {
  return Boolean(process.env.DATABASE_URL);
}

export type RankedTeam = TeamRecord & TeamStats & { rank: number };

/**
 * Fetch all active teams (optionally filtered by type) with computed stats.
 * Returns teams sorted by wins desc, then winRate desc (tie-break).
 * Returns [] in in-memory mode (no DATABASE_URL).
 */
export async function listAllTeamsWithStats(
  type?: "solo" | "duo"
): Promise<RankedTeam[]> {
  if (!hasDatabaseUrl()) return [];

  // Single query: fetch all active teams with members
  const teams = await prisma.team.findMany({
    where: { status: "active", ...(type ? { type } : {}) },
    include: { members: true },
    orderBy: { createdAt: "asc" },
  });

  if (teams.length === 0) return [];

  const teamIds = teams.map((t) => t.id);

  // Fetch ALL matches for these teams — no .take() (per pitfall 1)
  const rawMatches = await prisma.match.findMany({
    where: {
      OR: [{ teamAId: { in: teamIds } }, { teamBId: { in: teamIds } }],
    },
    orderBy: { playedAt: "desc" },
  });

  const allMatches: MatchRecord[] = rawMatches.map((m) => ({
    id: m.id,
    teamAId: m.teamAId,
    teamBId: m.teamBId,
    winnerTeamId: m.winnerTeamId,
    loserTeamId: m.teamAId === m.winnerTeamId ? m.teamBId : m.teamAId,
    playedAt: m.playedAt.toISOString(),
    note: m.note,
  }));

  // Compute stats per team then attach to team record
  const ranked = teams.map((team) => {
    const stats = computeTeamStats(team.id, allMatches);
    const teamRecord: TeamRecord = {
      id: team.id,
      name: team.name,
      type: team.type,
      status: team.status as "active" | "archived",
      createdBy: team.createdBy,
      createdAt: team.createdAt.toISOString(),
      updatedAt: team.updatedAt.toISOString(),
      members: team.members.map((m) => ({
        userId: m.userId,
        joinedAt: m.joinedAt.toISOString(),
      })),
    };
    return { ...teamRecord, ...stats, rank: 0 };
  });

  // Sort: primary wins desc, secondary winRate desc (D-03)
  ranked.sort((a, b) =>
    b.wins !== a.wins ? b.wins - a.wins : b.winRate - a.winRate
  );

  // Assign 1-indexed rank
  ranked.forEach((team, i) => { team.rank = i + 1; });

  return ranked;
}
```
